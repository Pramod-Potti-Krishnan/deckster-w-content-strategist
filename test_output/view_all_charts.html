Scatter Plot PYTHON
Ad Spend vs Revenue
import matplotlib.pyplot as plt import numpy as np # Data points x = [10000, 20000, 30000, 40000, 50000] y = [50000, 80000, 110000, 130000, 150000] # Create scatter plot fig, ax = plt.subplots(figsize=(10, 6)) ax.scatter(x, y, alpha=0.6, s=100, color='#667eea') ax.set_xlabel('Ad Spend ($)') ax.set_ylabel('Revenue ($)') ax.set_title('Ad Spend vs Revenue Correlation') ax.grid(True, alpha=0.3) # Add trend line z = np.polyfit(x, y, 1) p = np.poly1d(z) ax.plot(x, p(x), "r--", alpha=0.8) plt.tight_layout() plt.show()
Heatmap PYTHON
Correlation Matrix
import matplotlib.pyplot as plt import seaborn as sns import numpy as np # Generate correlation matrix data = np.random.rand(5, 5) data = (data + data.T) / 2 # Make symmetric np.fill_diagonal(data, 1) # Diagonal = 1 # Create heatmap plt.figure(figsize=(10, 8)) sns.heatmap(data, annot=True, fmt='.2f', cmap='coolwarm', center=0, square=True, linewidths=1, cbar_kws={"shrink": 0.8}) plt.title('Correlation Matrix Heatmap') plt.tight_layout() plt.show()
Histogram PYTHON
Age Distribution
import matplotlib.pyplot as plt import numpy as np # Generate age distribution data np.random.seed(42) ages = np.random.normal(35, 10, 1000) ages = np.clip(ages, 18, 65) # Create histogram plt.figure(figsize=(10, 6)) plt.hist(ages, bins=30, edgecolor='black', alpha=0.7, color='#667eea') plt.xlabel('Age') plt.ylabel('Frequency') plt.title('Customer Age Distribution') plt.grid(True, alpha=0.3, axis='y') plt.axvline(ages.mean(), color='red', linestyle='dashed', linewidth=2, label=f'Mean: {ages.mean():.1f}') plt.legend() plt.tight_layout() plt.show()
Waterfall Chart PYTHON
Profit Breakdown
import matplotlib.pyplot as plt import numpy as np # Waterfall data categories = ['Starting', 'Sales', 'Marketing', 'Operations', 'R&D', 'Final'] values = [100, 50, -20, -10, -5, 0] values[-1] = sum(values[:-1]) # Calculate positions cumulative = np.cumsum(values[:-1]) starts = np.concatenate([[0], cumulative[:-1]]) # Create waterfall fig, ax = plt.subplots(figsize=(10, 6)) for i, (cat, val, start) in enumerate(zip(categories[:-1], values[:-1], starts)): color = 'green' if val >= 0 else 'red' ax.bar(i, val, bottom=start, color=color, edgecolor='black') # Final bar ax.bar(len(categories)-1, values[-1], color='blue', edgecolor='black') ax.set_xticks(range(len(categories))) ax.set_xticklabels(categories) ax.set_ylabel('Profit ($k)') ax.set_title('Profit Waterfall Analysis') ax.grid(True, alpha=0.3, axis='y') plt.tight_layout() plt.show()
